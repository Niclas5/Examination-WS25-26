#Funktion wird definiert, um string-Einträge in numbers-Einträge umzuwandeln.
def clean_price(price_str):
    """Diese Funktion wandelt string in numeric um."""

    if pd.isna(price_str):
        return np.nan
    price_str_clean = str(price_str).replace("$", "").replace(',', '')
    try:
        price_number = float(price_str_clean)
        if price_number > 0:
            return price_number
        else:
            return np.nan
    except ValueError:
        return np.nan

def trimmed_mean(x, pct):
    """This function calculates the trimmed mean of a given array x, removing the lowest and highest pct-points of values."""
    if not (isinstance(pct, (int, float)) and 0 <= pct < 50): 
        raise ValueError("pct must be a number between 0 and 50")
    try:
        x = np.asarray(x, dtype=float)
    except (TypeError, ValueError):
        raise TypeError("x must be an array-like object containing numeric values")
    if x.ndim != 1:
        raise ValueError("x must be one-dimensional")
    bound = int(len(x) * pct / 100)
    if bound == 0:
        return np.mean(x)
    if 2*bound >= len(x):
        raise ValueError("pct too large for the size of x")
    x = x[~np.isnan(x)]
    x = np.sort(x)
    mean_trimmed = x[bound:-bound]
    return np.mean(mean_trimmed)

# We import re to check for and remove inconsistent whitespaces. 

import re

bzr_names = [feature["properties"]["BZR_NAME"] for feature in berlin_geojson["features"]]

geo_df = pd.DataFrame({"region_geo": bzr_names})

airbnb_regions = listings_1_9["neighbourhood_cleansed"].unique()

def clean_region(name):

    name = name.lower()
    name = name.replace("str.", "strasse")
    name = name.replace("-", " ")
    name = name.replace("ß", "ss")
    name = name.replace(" / ", "/")
    name = re.sub(r"\s+", " ", name)
    name = name.strip()

    name = name.split("/")[0]
    name = re.sub(r"(west \d+).*", r"\1", name)
    name = re.sub(r"(nord \d+).*", r"\1", name)
    name = re.sub(r"(ost \d+).*", r"\1", name)

    return name

geo_df["clean_name"] = geo_df["region_geo"].apply(clean_region)
airbnb_df["clean_name"] = airbnb_df["region_airbnb"].apply(clean_region)

# Checking for perfect matches after cleaning function

auto_matches = set(geo_df["clean_name"]) & set(airbnb_df["clean_name"])

mapping_auto = {name: name for name in auto_matches}

# Connecting the pairs of" AirBnB names => geojson names" corresponding to the perfect matches in the mapping_auto dictionary.

mapping_auto_v2 = {airbnb_df.loc[airbnb_df["clean_name"]==name]["region_airbnb"].iloc[0]: geo_df.loc[geo_df["clean_name"]==name]["region_geo"].iloc[0] for name in auto_matches}

# Checking for remaining unmatched regions
